\begin{modnl}[creators=hang]{complement-relative}{cns}
	\begin{definition}[id=complement-relative.def]
	如果$a$和$b$是\trefi[sets]{集合},则  
	$a$在$b$中的\defi{complement-relative}[相对补集]，是由所有在$b$
	\trefi[sets]{集合}的元素但不在$a$的元素所组成的\trefi[sets]{集合}.
	\end{definition}
\end{modnl}
