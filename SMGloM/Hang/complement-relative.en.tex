\begin{modnl}[creators=hang]{complement-relative}{en}
	\begin{definition}[id=C-relative.def]
	If $a$ and $b$ are \trefi[sets]{sets}, then the 
	\defi{complement-relative} of $a$ in $b$, is the 
	\trefi[sets]{set} of elements in $b$, but not in $a$.
	\end{definition}
\end{modnl}
