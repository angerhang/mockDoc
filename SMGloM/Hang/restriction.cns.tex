\begin{modnl}[creator=hang]{restriction}{cns}
	\begin{definition}[id=restriction.def]
	假设 $A$ 是在集合\trefi[set]{集合}$B$上的\trefi[relation]{关系}，
	那么$A$对$B$\的defi[restriction]限制，是被剔除所有不属于$A$的
	元素的$B$\trefi[set]{集合}。
	\end{definition}
\end{modnl}
