\begin{modnl}[creator=hang]{restriction}{en}
	\begin{definition}[id=restriction.def]
	Let $A$ be the \trefi[relation]{relation} on \trefi[set]{set} $B$,
	then we say the \defi[restriction]{restriction} of $A$ on $B$ is the set $B$
	that takes out all the elements that do not belong to $A$.
	\end{definition}
\end{modnl}
