\begin{modnl}[creators=Jinbo]{multiset}{cns}
	\begin{definition}[id=multiset.def]
		\defi[multiset]{多重集} 是集合概念的推广。\defi[multiset]{多重集} 允许同一个元素出现多次。
	\end{definition}

	\begin{definition}
		一个元素的 \defi[multiplicity]{重数} (或 \defi[multiplicity]{重复度}) 是指这个元素在此 \defi{multiset}[多重集] 中出现的次数。
	\end{definition}
\end{modnl}
