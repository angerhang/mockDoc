\begin{modnl}[creators=Jinbo]{multiset}{cnt}
	\begin{definition}[id=multiset.def]
		\defi[multiset]{多重集} 是集合概念的推廣。\defi[multiset]{多重集} 允許同一個元素出現多次。
	\end{definition}

	\begin{definition}
		一個元素的 \defi[multiplicity]{重數} (或 \defi[multiplicity]{重復度}) 是指這個元素在此 \defi{multiset}[多重集] 中出現的次數。
	\end{definition}
\end{modnl}
