\begin{modnl}[creators=Jinbo]{multiset}{en}

\begin{definition}[id=multiset.def]
	A \defi{multiset} is a generalization of the concept of a set, whereas
\defi{multiset} allows multiple instances for every elements. 
\end{definition}

\begin{definition}
	The \defi{multiplicity} of an element is the number of instances 
of the element in a specific \defi{multiset}.
\end{definition}

\end{modnl}
